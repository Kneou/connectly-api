TEST 1: Create Post
POST /posts
Body:
{
  "content": "Hello TA",
  "privacy": "public",
  "userId": 1
}
Expected: 201 Created

GET /posts?page=1

POST /posts/1/comment
Body:
{
  "content": "Nice post"
}
Expected: 201 Created

POST /posts/1/like
Expected: Post liked

DELETE /posts/1

role: admin
